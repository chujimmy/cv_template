% Override name box: 
% - ensure name is aligned with top 
% - ensure position aligned with bottom 
% - use normal text for position
\xpatchcmd{\makecvhead}{\usebox{\makecvheadnamebox}}{
    \begin{minipage}[b]{\makecvheadnamewidth}
        \newsavebox{\nameBox}
        \savebox{\nameBox}{\namestyle{\@firstname\ \@lastname}}
        \hasBelowBaselineName{\@firstname\ \@lastname}

        \if@left\raggedright\fi%
        \if@right\raggedleft\fi%
        \namestyle{\@firstname\ \@lastname}
        \ifthenelse{\equal{\@title}{}}
            {}
            {
                \ifthenelse{\nameContainsBelowBaselineChar>0}
                    {\\[ 6em - \the\ht\nameBox - \baselineskip ]}
                    {\\[ 6em - \the\ht\nameBox ]}
                \titlestyle{\textnormal{\@title}}
            }
    \end{minipage}
}{}{}

% This function is needed in order to keep the name aligned with the top left
% and the position title aligned with the last line of the details block
\newcommand*{\hasBelowBaselineName}[1]{
    % This list is not necessary complete and should cover basic descenders (letters going below baseline)
    % Feel free to upgrade this list to ensure all cases are handled
    \def\charList{g,j,p,q,y}
    \def\nameContainsBelowBaselineChar{0}

    \foreach \char in \charList {
        \IfSubStr{#1}{\char}
            {\gdef\nameContainsBelowBaselineChar{1}}
            {}
    }
}
